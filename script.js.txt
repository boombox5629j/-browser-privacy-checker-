// Wait for DOM to load
document.addEventListener('DOMContentLoaded', async () => {
    const loading = document.getElementById('loading');
    const results = document.getElementById('results');
    
    // Collect browser data
    const browserData = collectBrowserData();
    
    // Calculate risk score
    const riskScore = calculateRiskScore(browserData);
    const riskLevel = getRiskLevel(riskScore);
    
    // Display results
    displayResults(browserData, riskScore, riskLevel);
    
    loading.classList.add('hidden');
    results.classList.remove('hidden');
});

function parseUserAgent(ua) {
    const lower = ua.toLowerCase();

    // Mobile detection
    if (lower.includes('iphone') || lower.includes('ipad')) {
        if (lower.includes('crios')) return 'Chrome iOS';
        return 'Safari iOS';
    }
    if (lower.includes('android')) {
        if (lower.includes('chrome')) return 'Chrome Android';
        return 'Android Browser';
    }

    // Desktop
    if (lower.includes('edg/')) return 'Edge';
    if (lower.includes('firefox')) return 'Firefox';
    if (lower.includes('chrome')) return 'Chrome';
    if (lower.includes('safari')) return 'Safari';

    return 'Unknown';
}

function parseUserAgent(ua) {
    if (ua.includes('Chrome')) return 'Chrome';
    if (ua.includes('Firefox')) return 'Firefox';
    if (ua.includes('Safari')) return 'Safari';
    if (ua.includes('Edge')) return 'Edge';
    return 'Unknown';
}

function calculateRiskScore(data) {
    let score = 0;
    
    // Cookies enabled? (+20)
    if (data.cookiesEnabled) score += 20;
    
    // Common screen sizes are more trackable (+15)
    if ([1366, 1920, 1536].includes(data.screenWidth)) score += 15;
    
    // Default language (+10)
    if (data.language === 'en-US' || data.language === 'en') score += 10;
    
    // No Do Not Track (+15)
    if (data.doNotTrack === 'unspecified' || !data.doNotTrack) score += 15;
    
    // More CPU cores = more unique (+10)
    if (data.cpuCores > 4) score += 10;
    
    return Math.min(score, 100);
}

function getRiskLevel(score) {
    if (score <= 30) return { label: 'Low', color: 'low' };
    if (score <= 60) return { label: 'Medium', color: 'medium' };
    return { label: 'High', color: 'high' };
}

function displayResults(data, score, risk) {
    // Risk meter
    document.getElementById('risk-label').textContent = `${risk.label} Tracking Risk`;
    document.getElementById('risk-label').className = risk.color;
    document.getElementById('risk-score').textContent = score;
    document.getElementById('risk-score').className = risk.color;
    document.getElementById('risk-fill').style.width = `${score}%`;
    document.getElementById('risk-fill').className = `risk-fill ${risk.color}`;
    
    // Device info
    document.getElementById('device-info').innerHTML = `
        <div class="info-item"><span>Browser:</span><span>${data.browser}</span></div>
        <div class="info-item"><span>Platform:</span><span>${data.platform}</span></div>

        <div class="info-item"><span>Do Not Track:</span><span>${data.doNotTrack}</span></div>
    `;
    
    // Screen info
    document.getElementById('screen-info').innerHTML = `
