// Canvas fingerprint
function generateCanvasFingerprint() {
    const canvas = document.getElementById('canvas-fp');
    const ctx = canvas.getContext('2d');
    
    ctx.fillStyle = '#f0f0f0';
    ctx.fillRect(0, 0, 200, 60);
    ctx.fillStyle = '#000';
    ctx.font = '14px Arial';
    ctx.fillText('Fingerprint test 🖼️', 2, 30);
}

// WebGL fingerprint
function getWebGLFingerprint() {
    const canvas = document.createElement('canvas');
    const gl = canvas.getContext('webgl') || canvas.getContext('experimental-webgl');
    if (!gl) return 'WebGL not supported';
    
    const debugInfo = gl.getExtension('WEBGL_debug_renderer_info');
    const vendor = debugInfo ? gl.getParameter(debugInfo.UNMASKED_VENDOR_WEBGL) : gl.getParameter(gl.VENDOR);
    const renderer = debugInfo ? gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL) : gl.getParameter(gl.RENDERER);
    
    return `${vendor} | ${renderer}`.replace(/\s+/g, ' ').trim();
}

// Show all fingerprints
function showFingerprints() {
    generateCanvasFingerprint();
    document.getElementById('webgl-info').textContent = getWebGLFingerprint();
}

function calculateRiskScore(data) {
    let score = 20; // Base
    if (data.cookiesEnabled) score += 20;
    if (data.cpuCores > 4) score += 15;
    if (data.language === 'en-US') score += 10;
    return Math.min(score + 35, 100); // Demo score
}

function getRiskLevel(score) {
    if (score <= 40) return { label: 'Low', color: 'low' };
    if (score <= 70) return { label: 'Medium', color: 'medium' };
    return { label: 'High', color: 'high' };
}
